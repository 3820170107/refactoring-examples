###

```
public class IntRange
{
  private int low, high;

  public IntRange(int low, int high)
  {
    this.low = low;
    this.high = high;
  }

  public bool Includes(int arg)
  {
    return arg >= low && arg <= high;
  }
  public void Grow(int factor)
  {
    high = high * factor;
  }
}
```

###

```
public class IntRange
{
  private int Low
  {
    get;
    set;
  }
  private int High
  {
    get;
    set;
  }

  public IntRange(int low, int high)
  {
    this.Low = low;
    this.High = high;
  }

  public bool Includes(int arg)
  {
    return arg >= Low && arg <= High;
  }
  public void Grow(int factor)
  {
    High = High * factor;
  }
}
```
