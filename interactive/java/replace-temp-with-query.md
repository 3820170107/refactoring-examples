###

```
class Product {
  // ...
  public double getPrice() {
    int basePrice = quantity * itemPrice;
    double discountFactor;
    if (basePrice > 1000) {
      discountFactor = 0.95;
    }
    else {
      discountFactor = 0.98;
    }
    return basePrice * discountFactor;
  }
}
```

###

```
class Product {
  // ...
  public double getPrice() {
    return basePrice() * discountFactor();
  }
  private int basePrice() {
    return quantity * itemPrice;
  }
  private double discountFactor() {
    if (basePrice() > 1000) {
      return 0.95;
    }
    else {
      return 0.98;
    }
  }
}
```
