```
class Product {
  // ...
  function getPrice() {
    return $this->basePrice() * $this->discountFactor();
  }
  private function basePrice() {
    return $this->quantity * $this->itemPrice;
  }
  private function discountFactor() {
    if ($this->basePrice() > 1000) {
      return 0.95;
    }
    else {
      return 0.98;
    }
  }
}
```
