      $result += $this->type->overdraftCharge($this);

  }

  // ...

    if ($this->isPremium()) {
      $result = 10;
      if ($account->getDaysOverdrawn() > 7) {
        $result += ($account->getDaysOverdrawn() - 7) * 0.85;
      }
      return $result;
    }
    else {
      return $account->getDaysOverdrawn() * 1.75;
    }
  }
}
